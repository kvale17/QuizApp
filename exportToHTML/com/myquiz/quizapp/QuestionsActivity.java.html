144      public void getQuestionList()
145      {
146  
147          questionList.clear();
148          loadingDialog.show();
149  
150          firestore.collection("QUIZ").document(subList.get(selected_sub_index).getId())
151                  .collection(gradesIDs.get(GradeNo).getId()).get()
152                  .addOnSuccessListener(new OnSuccessListener<QuerySnapshot>() {
153                      @Override
154                      public void onSuccess(QuerySnapshot queryDocumentSnapshots) {
155  
156                          Map<String, QueryDocumentSnapshot> docList = new ArrayMap<>();
157  
158                          for(QueryDocumentSnapshot doc : queryDocumentSnapshots)
159                          {
160                              docList.put(doc.getId(),doc);
161                          }
162  
163                          QueryDocumentSnapshot quesListDoc = docList.get("QUESTIONS_LIST");
164  
165                          String count = quesListDoc.getString("COUNT");
166  
167                          for(int i = 0; i<Integer.valueOf(count);i++)
168                          {
169  
170                              String questID = quesListDoc.getString("Q" + String.valueOf(i+1) + "_ID");
171  
172                              QueryDocumentSnapshot quesDoc = docList.get(questID);
173  
174                              questionList.add(new Question(
175                                      quesDoc.getString("QUESTION"),
176                                      quesDoc.getString("A"),
177                                      quesDoc.getString("B"),
178                                      quesDoc.getString("C"),
179                                      quesDoc.getString("D"),
180                                      Integer.valueOf(quesDoc.getString("ANSWER"))
181                              ));
182  
183                          }
184  
185  
186                          setQuestion();
187  
188                          loadingDialog.dismiss();
189  
190  
191                      }
192                  }).addOnFailureListener(new OnFailureListener() {
193              @Override
194              public void onFailure(@NonNull Exception e) {
195                  Toast.makeText(QuestionsActivity.this, e.getMessage(), Toast.LENGTH_SHORT).show();
196                  loadingDialog.dismiss();
197              }
198          });
199  
200  
201      }

219      private void checkAnswer(int selectedOption, View view)
220      {
221  
222              if (selectedOption == questionList.get(questionNum).getCorrectAns())
223              {
224                  //Right Answer
225  
226                  score++;
227                  ((Button)view).setBackgroundTintList(ColorStateList.valueOf(Color.GREEN));
228  
229  
230              } else
231  
232                  {
233                  //Wrong Answer
234  
235                      ((Button)view).setBackgroundTintList(ColorStateList.valueOf(Color.RED));
236  
237                      switch (questionList.get(questionNum).getCorrectAns())
238                      {
239                          case 1:
240                              Option1.setBackgroundTintList(ColorStateList.valueOf(Color.GREEN));
241                              break;
242                          case 2:
243                              Option2.setBackgroundTintList(ColorStateList.valueOf(Color.GREEN));
244                              break;
245                          case 3:
246                              Option3.setBackgroundTintList(ColorStateList.valueOf(Color.GREEN));
247                              break;
248                          case 4:
249                              Option4.setBackgroundTintList(ColorStateList.valueOf(Color.GREEN));
250                              break;
251                          default:
252                              break;
253                      }
254              }
255  
256              Handler handler = new Handler();
257              handler.postDelayed(new Runnable() {
258                  @Override
259                  public void run() {
260                      changeQuestion();
261                  }
262              }, 3000);
263  
264  
265      }

267      @SuppressLint("SetTextI18n")
268      private void changeQuestion()
269      {
270  
271          if(questionNum<(questionList.size()-1))
272          {
273              //change question
274  
275              questionNum++;
276  
277              changeAnim(question, 0, 0);
278              changeAnim(Option1, 0,1);
279              changeAnim(Option2, 0,2);
280              changeAnim(Option3, 0,3);
281              changeAnim(Option4, 0,4);
282  
283              count.setText((questionNum + 1) + "/" + questionList.size());
284  
285  
286          }
287          else
288          {
289              //Go to score activity
290              Intent intent = new Intent(QuestionsActivity.this, ScoreActivity.class);
291              intent.putExtra("SCORE", score + "/" + questionList.size());
292              intent.setFlags(Intent.FLAG_ACTIVITY_NEW_TASK | Intent.FLAG_ACTIVITY_CLEAR_TASK);
293              startActivity(intent);
294          }
295          selectedOption = 0;
296          btnSubmit.setClickable(true);
297          Option1.setClickable(true);
298          Option2.setClickable(true);
299          Option3.setClickable(true);
300          Option4.setClickable(true);
301  
302  
303      }

305      private void changeAnim(final View view, final int value, final int viewNum)
306      {
307          view.animate().alpha(value).scaleX(value).scaleY(value).setDuration(500)
308                  .setStartDelay(100).setInterpolator(new DecelerateInterpolator())
309                  .setListener(new Animator.AnimatorListener() {
310                      @Override
311                      public void onAnimationStart(Animator animation) {
312  
313                      }
314  
315                      @Override
316                      public void onAnimationEnd(Animator animation) {
317                         if(value == 0)
318                         {
319                             switch (viewNum)
320                             {
321                                 case 0:
322                                     ((TextView)view).setText((questionList.get(questionNum).getQuestion()));
323                                     break;
324                                 case 1:
325                                     ((Button)view).setText(questionList.get(questionNum).getOptionA());
326                                     break;
327                                 case 2:
328                                     ((Button)view).setText(questionList.get(questionNum).getOptionB());
329                                     break;
330                                 case 3:
331                                     ((Button)view).setText(questionList.get(questionNum).getOptionC());
332                                     break;
333                                 case 4:
334                                     ((Button)view).setText(questionList.get(questionNum).getOptionD());
335                                     break;
336                                 default:
337                                     break;
338                             }
339  
340                             if(viewNum != 0)
341                             {
342                                 ((Button)view).setBackgroundTintList(ColorStateList.valueOf(Color.parseColor("#FFFFFF")));
343                             }
344  
345                             changeAnim(view,1,viewNum);
346                         }
347  
348                      }
349  
350                      @Override
351                      public void onAnimationCancel(Animator animation) {
352  
353                      }
354  
355                      @Override
356                      public void onAnimationRepeat(Animator animation) {
357  
358                      }
359                  });
360      }
